public int peopleAwareOfSecret(int n, int delay, int forget) {
    int mod = (int) 1e9 + 7;
    long[] f = new long[n + 1];
    long[] preSum = new long[n + 1];
    f[1] = 1;
    preSum[1] = 1;

        int l = Math.max(0, i - forget);

public int countPaths(int[][] grid) {
    int ans = 0;
    this.grid = grid;
    this.m = grid.length;

}
